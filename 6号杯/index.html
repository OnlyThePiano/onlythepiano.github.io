	private void changeText(File fileText, String target, String src)
	{
		BufferedReader br = null;
		PrintWriter pw = null;
		ArrayList list = new ArrayList();
		
		try {
			br = new BufferedReader(new FileReader(fileText));
			String str = null;
			while ((str=br.readLine()) != null)
			{
			
				str = str.replaceAll(target, src);
				System.out.println(str);
				list.add(str);
					
				
			}

			
			pw = new PrintWriter(new FileWriter(fileText));
			
			for (int i=0; i<list.size(); i++)
			{
				String str1 = (String)list.get(i);
				System.out.println("doing1");
				pw.println(str1);
				
			}
			System.out.println("Done, Finish");
		
			pw.flush();
			
			
		} catch (FileNotFoundException e) {

            e.printStackTrace();
            
        } catch (IOException e) {

            e.printStackTrace();
            
        }

	}
